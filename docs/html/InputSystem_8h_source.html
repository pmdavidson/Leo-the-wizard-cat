562 {
563 switch (type)
564 {
565 case SpellType::Fire:
566 {
567 static std::mt19937 rng(std::random_device{}());
568 std::uniform_int_distribution<int> dist(1, 2);
569 std::string soundName = "fire_cast_" + std::to_string(dist(rng));
570 soundManager.PlaySound(soundName);

572 break;
573 case SpellType::Water:
574 soundManager.PlaySound("water_cast");
575 break;
576 case SpellType::Wind:
