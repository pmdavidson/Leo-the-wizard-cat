27 bool Run(Scene<Components...> &scene) override
28 {
29 auto &entityManager = scene.GetEntityManager();
30 auto &soundManager = scene.GetSoundManager();
31 float deltaTime = 1.0f / 60.0f;
32
33 for (auto it = entityManager.begin(); it != entityManager.end(); ++it)
34 {
35 if (!it->isActive())
36 continue;
37 EntityID entityId = it->getID();
38
39 // Only process entities with SpellComponent and InputComponent
40 if (!entityManager.template HasComponent<SpellComponent>(entityId) ||
41 !entityManager.template HasComponent<InputComponent>(entityId))
42 continue;
43
44 // Skip spell processing if player is dead or dying
45 if (entityManager.template HasComponent<HpComponent>(entityId))
46 {
47 auto &hpComp = entityManager.template GetComponent<HpComponent>(entityId);
48 if (!hpComp.isAlive || hpComp.isDying)
49 {
50 continue; // Skip all spell processing during death
51 }
52 }
53
54 // Update rock shield based on selected spell
55 if (entityManager.template HasComponent<HpComponent>(entityId) &&
56 entityManager.template HasComponent<SpellComponent>(entityId))
57 {
58 auto &hpComp = entityManager.template GetComponent<HpComponent>(entityId);
59 auto &spellComp = entityManager.template GetComponent<SpellComponent>(entityId);
60
61 // Get previous spell
62 SpellType prevSpell = spellComp.selectedSpell;
63 if (mPrevSelectedSpell.find(entityId) != mPrevSelectedSpell.end())
64 {
65 prevSpell = mPrevSelectedSpell[entityId];
66 }
67
68 // Check if we just switched TO rock
69 bool justSwitchedToRock = (spellComp.selectedSpell == SpellType::Earth) &&
70 (prevSpell != SpellType::Earth);
71
72 // Activate rock shield only when switching to rock
73 if (justSwitchedToRock)
74 {
75 hpComp.hasRockShield = true;
76 }
77 // Deactivate shield when switching away from rock
78 else if (spellComp.selectedSpell != SpellType::Earth && hpComp.hasRockShield)
79 {
80 hpComp.hasRockShield = false;
81 }
82
83 // Check if we just switched TO water
84 bool justSwitchedToWater = (spellComp.selectedSpell == SpellType::Water) &&
85 (prevSpell != SpellType::Water);
86
87 // Activate rock shield only when switching to rock
88 if (justSwitchedToWater)
89 {
90 hpComp.canHeal = true;
91 }
92 // Deactivate shield when switching away from rock
93 else if (spellComp.selectedSpell != SpellType::Water && hpComp.canHeal)
94 {
95 hpComp.canHeal = false;
96 }
97
98 // Track current spell for next frame
99 mPrevSelectedSpell[entityId] = spellComp.selectedSpell;
100 }
101
102 auto &spellComp = entityManager.template GetComponent<SpellComponent>(entityId);
103 auto &inputComp = entityManager.template GetComponent<InputComponent>(entityId);
104
105 // Update cooldowns
106 UpdateCooldowns(spellComp, deltaTime);
107
108 // Update facing direction based on movement input
109 UpdateFacingDirection(spellComp, inputComp);
110
111 // Process element selection (1, 2, 3, 4 keys)
112 ProcessElementSelection(spellComp, inputComp, entityId, soundManager);
113
114 // Process spell casting (E key or left mouse button equivalent)
115 ProcessSpellCast(scene, entityId, spellComp, inputComp, soundManager);
116 }
117
118 return true;
119 }

169 // Update facing direction based on input
170 void UpdateFacingDirection(SpellComponent &spellComp, const InputComponent &inputComp)
171 {
172 sf::Keyboard::Scancode scancodeA = sf::Keyboard::delocalize(sf::Keyboard::Key::A);
173 sf::Keyboard::Scancode scancodeD = sf::Keyboard::delocalize(sf::Keyboard::Key::D);
174
175 bool movingLeft = (scancodeA != sf::Keyboard::Scan::Unknown &&
176 static_cast<size_t>(scancodeA) < sf::Keyboard::ScancodeCount)
177 ? inputComp.keydown[static_cast<size_t>(scancodeA)]
178 : false;
179
180 bool movingRight = (scancodeD != sf::Keyboard::Scan::Unknown &&
181 static_cast<size_t>(scancodeD) < sf::Keyboard::ScancodeCount)
182 ? inputComp.keydown[static_cast<size_t>(scancodeD)]
183 : false;
184
185 if (movingLeft && !movingRight)
186 spellComp.facingDirection = -1;
187 else if (movingRight && !movingLeft)
188 spellComp.facingDirection = 1;
189 }

191 // Process element selection keys (1, 2, 3, 4)
192 void ProcessElementSelection(SpellComponent &spellComp, const InputComponent &inputComp,
193 EntityID entityId, SoundManager &soundManager)
194 {
195 sf::Keyboard::Scancode scancode1 = sf::Keyboard::delocalize(sf::Keyboard::Key::Num1);
196 sf::Keyboard::Scancode scancode2 = sf::Keyboard::delocalize(sf::Keyboard::Key::Num2);
197 sf::Keyboard::Scancode scancode3 = sf::Keyboard::delocalize(sf::Keyboard::Key::Num3);
198 sf::Keyboard::Scancode scancode4 = sf::Keyboard::delocalize(sf::Keyboard::Key::Num4);
199
200 auto &prevStates = mPrevSelectKeyStates[entityId];
201
202 std::array<sf::Keyboard::Scancode, 4> selectKeys = {scancode1, scancode2, scancode3, scancode4};
203 std::array<SpellType, 4> spellTypes = {SpellType::Fire, SpellType::Water, SpellType::Wind, SpellType::Earth};
204 std::array<std::string, 4> spellNames = {"Fire", "Water", "Wind", "Rock"}; // Debug
205
206 for (size_t i = 0; i < 4; ++i)
207 {
208 bool keyPressed = (selectKeys[i] != sf::Keyboard::Scan::Unknown &&
209 static_cast<size_t>(selectKeys[i]) < sf::Keyboard::ScancodeCount)
210 ? inputComp.keydown[static_cast<size_t>(selectKeys[i])]
211 : false;
212
213 // Switch element on key press (not hold), only if different element and can switch
214 if (keyPressed && !prevStates[i])
215 {
216 if (spellComp.selectedSpell != spellTypes[i] && CanSwitch(spellComp))
217 {
218 spellComp.selectedSpell = spellTypes[i];
219 StartSwitchCooldown(spellComp);
220 PlaySwitchSound(spellTypes[i], soundManager);
221 std::cout << "Element switched to: " << spellNames[i] << " (key " << (i + 1) << ")" << std::endl;
222 }
223 }
224
225 prevStates[i] = keyPressed;
226 }

252 // Create a projectile entity for the selected spell
253 void CastSpell(Scene<Components...> &scene, EntityID casterId,
254 SpellComponent &spellComp, SoundManager &soundManager)
255 {
256 auto &entityManager = scene.GetEntityManager();
257 [[maybe_unused]] auto &spriteManager = scene.GetSpriteManager();
258
259 // Get caster position and collision box
260 if (!entityManager.template HasComponent<LocationComponent>(casterId) ||
261 !entityManager.template HasComponent<CollisionComponent>(casterId))
262 return;
263
264 auto &casterLoc = entityManager.template GetComponent<LocationComponent>(casterId);
265 auto &casterCollision = entityManager.template GetComponent<CollisionComponent>(casterId);
266 const SpellProperties &props = GetSelectedProperties(spellComp);
267
268 // Calculate spawn position
269 // Spawn in front of player based on facing direction
270 float spawnOffsetX = spellComp.facingDirection * 30.0f; // Spawn 30 pixels in front
271
272 // Align bottom of spell AABB with bottom of player AABB
273 // Player collision bottom = casterLoc.position.y + casterCollision.localBounds.height
274 // Spell collision bottom = spellPos.y + 22.0f (all spells are 28x22)
275 // We want: spellPos.y + 22.0f = casterLoc.position.y + casterCollision.localBounds.height
276 // So: spellPos.y = casterLoc.position.y + casterCollision.localBounds.height - 22.0f
277 float playerBottom = casterLoc.position.y + casterCollision.localBounds.height;
278 Point2D spawnPos = {
279 casterLoc.position.x + spawnOffsetX,
280 playerBottom - 22.0f // 28x22 collision box (height is 22)
281 };
282
283 // Fixed spell box size used for both collision and spawn check
284 Rect collisionBounds(0.0f, 0.0f, 28.0f, 22.0f);
285
286 // Create projectile entity
287 std::string spellName = GetSpellName(spellComp.selectedSpell);
288 EntityID projectileId = entityManager.CreateEntity(spellName);
289
290 // Add LocationComponent
291 entityManager.template AddComponent<LocationComponent>(projectileId, LocationComponent(spawnPos));
292
293 // Add MovementComponent with velocity based on facing direction
294 MovementComponent movement;
295 movement.velocity = Point2D(spellComp.facingDirection * props.speed, 0.0f);
296 entityManager.template AddComponent<MovementComponent>(projectileId, movement);
297
298 // Add ProjectileComponent
299 ProjectileComponent projectile;
300 projectile.spellType = spellComp.selectedSpell;
301 projectile.damage = props.damage;
302 projectile.lifetime = props.lifetime;
303 projectile.maxLifetime = props.lifetime;
304 projectile.ownerEntityId = casterId;
305 projectile.active = true;
306 projectile.explosionFrames = props.explosionFrames;
307 projectile.explosionSize = props.explosionSize;
308 entityManager.template AddComponent<ProjectileComponent>(projectileId, projectile);
309
310 // Add CollisionComponent - all spells are 28x22
311 CollisionComponent collision(collisionBounds, false);
312 entityManager.template AddComponent<CollisionComponent>(projectileId, collision);
313
314 // Add SpriteComponent
315 // All spells use 28x22 bounds to match collision box
316 SpriteComponent sprite;
317 sprite.spriteId = props.spriteId;
318 sprite.bounds = Rect(0.0f, 0.0f, 28.0f, 22.0f);
319 sprite.inWorldSpace = true;
320 sprite.layer = 2; // Spells should be on layer 2 (just above world)
321 sprite.flipX = (spellComp.facingDirection < 0);
322 entityManager.template AddComponent<SpriteComponent>(projectileId, sprite);
323
324 // Add AnimationComponent for flying animation
325 if (!props.flyingFrames.empty())
326 {
327 AnimationComponent flyingAnim;
328 flyingAnim.animations["flying"] = props.flyingFrames;
329 flyingAnim.currentAnimation = "flying";
330 flyingAnim.frameDuration = 0.05f;
331 flyingAnim.playing = true;
332 flyingAnim.looping = true;
333 entityManager.template AddComponent<AnimationComponent>(projectileId, flyingAnim);
334 }
335
336 // Start cast cooldown
337 StartCastCooldown(spellComp);
338
339 // Play cast sound
340 PlayCastSound(spellComp.selectedSpell, soundManager);
341 }

396 break;
